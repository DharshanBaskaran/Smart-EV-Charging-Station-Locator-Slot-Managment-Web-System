3. Coding (Key Code Snippets)

3.1 Haversine Distance Calculation (server.js)

function haversineKm(lat1, lon1, lat2, lon2) {
  const R = 6371; // Earth's radius in km
  const dLat = (lat2 - lat1) * Math.PI / 180;
  const dLon = (lon2 - lon1) * Math.PI / 180;
  const a =
    Math.sin(dLat / 2) ** 2 +
    Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
    Math.sin(dLon / 2) ** 2;
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
}

3.2 Real-Time Port Status Simulation (server.js)

function getPortStatus(portId) {
  const t    = Math.floor(Date.now() / 10000);
  const seed = (portId + t).split('').reduce((a, c) => a + c.charCodeAt(0), 0);
  return {
    functional:  seed % 10 > 1,  // 80% chance functional
    occupancy:   (seed + t) % 4 === 0 ? 'occupied' : 'free',
    lastUpdated: new Date().toISOString(),
  };
}

3.3 Password Hashing & Token-Based Authentication (server.js)

function hashPassword(password) {
  return crypto.createHash('sha256').update(password).digest('hex');
}

// Login Route
app.post('/api/auth/login', async (req, res) => {
  const { username, password } = req.body || {};
  const user = await User.findOne({ username: username.toLowerCase() });
  if (!user || user.passwordHash !== hashPassword(password))
    return res.status(401).json({ error: 'Invalid username or password' });

  const token = crypto.randomBytes(32).toString('hex');
  sessions[token] = { userId: user.userId, role: user.role };
  const { passwordHash, ...safe } = user.toObject();

3.4 Range-Based Station Filtering (server.js)

app.get('/api/stations', async (req, res) => {
  let stations = await Station.find({});
  const lat     = parseFloat(req.query.lat);
  const lng     = parseFloat(req.query.lng);
  const rangeKm = parseFloat(req.query.rangeKm);

  let result = stations.map(s => ({ ...s.toObject(), id: s.stationId }));

  if (Number.isFinite(lat) && Number.isFinite(lng) &&
      Number.isFinite(rangeKm) && rangeKm > 0) {
    result = result.filter(s => haversineKm(lat, lng, s.lat, s.lng) <= rangeKm);
    result.sort((a, b) =>
      haversineKm(lat, lng, a.lat, a.lng) - haversineKm(lat, lng, b.lat, b.lng)
    );
  }
  res.json(result);
});

3.5 Trip Planner – Charging Stop Calculation (app.js)

async function calculateTrip(startLatLng, destLatLng, startName, destName) {
  const rangeKm   = parseFloat(rangeInput.value) || 80;
  const totalDist  = startLatLng.distanceTo(destLatLng) / 1000;
  const step       = rangeKm * 0.8; // 80% of battery range per segment
  const numStops   = Math.floor(totalDist / step);

  let stops = [];
  for (let i = 1; i <= numStops; i++) {
    const fraction = (i * step) / totalDist;
    const idealLat = startLatLng.lat + dLat * fraction;
    const idealLng = startLatLng.lng + dLng * fraction;
    const stations = await fetchStations({ lat: idealLat, lng: idealLng, rangeKm: 30 });
    // Select nearest non-duplicate station
    const best = stations.find(st =>
      !stops.some(s => L.latLng(s.lat, s.lng).distanceTo(L.latLng(st.lat, st.lng)) / 1000 < 10)
    );
    if (best) stops.push(best);
  }
}

3.6 Reservation Conflict Check (server.js)

app.post('/api/reservations', async (req, res) => {
  const { portId, startTime, endTime } = req.body;
  // Conflict check in DB
  const conflict = await Reservation.findOne({
    portId,
    status: 'confirmed',
    startTime: new Date(startTime),
  });
  if (conflict) return res.status(409).json({ error: 'Slot already reserved' });

  const reservation = await Reservation.create({
    reservationId: `res_${Date.now()}_${Math.random().toString(36).slice(2, 7)}`,
    userId, portId,
    startTime: new Date(startTime),
    endTime:   new Date(endTime),
    status:    'confirmed',
  });
  res.status(201).json({ ...reservation.toObject(), id: reservation.reservationId });
});

6. Implementation Issues

# Issue Description Resolution
1 Simulated Port Status Real-time port status is simulated using a deterministic algorithm (seed-based) instead of actual IoT sensor data. Status changes every ~10 seconds automatically. Acceptable for proof-of-concept. Architecture is IoT-ready — can replace getPortStatus() with MQTT/webhook integration in production.
2 In-Memory Session Store User sessions (token → userId mapping) are stored in server memory using a JavaScript object. Sessions are lost when the server restarts. Sufficient for development and demo. For production, sessions should be stored in Redis or MongoDB with token expiry.
3 Straight-Line Distance for Trip Planning The trip planner uses Haversine (great-circle) distance between origin and destination rather than actual road distance. This may suggest more or fewer stops than actually needed. For accurate road-based routing, integration with Google Directions API or OSRM is required as a future enhancement.
4 No HTTPS in Development The application runs on HTTP during development, meaning tokens are transmitted in plain text over the network. HTTPS must be enabled in production using SSL/TLS certificates (e.g., via Let's Encrypt or cloud load balancer).
5 CORS Open Policy The server uses cors() with no origin restrictions, allowing requests from any domain. For production, CORS should be configured to allow only the specific frontend domain.
6 Geocoding Rate Limits The Nominatim API (OpenStreetMap) used for geocoding in the trip planner has a usage policy of 1 request per second and no commercial use without permission. For high-traffic production use, a paid geocoding service (Google Geocoding API, Mapbox) should replace Nominatim.
